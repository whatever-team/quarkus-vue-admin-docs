├── common            // 通用功能
│       └── base                          // 基类

│       └── util                          // 工具类
├── handler           // 全局处理器
├── lifecycle         // 生命周期
├── task              // 定时任务
├── web               // Web请求
│       └── business                      // 业务代码
│       └── common                        // 通用请求
│       └── system                        // 系统请求
│               └── dto                      // DTO类
│               └── entity                   // 数据实体类
│               └── param                    // 入参类
│               └── resource                 // 资源请求

│       └── manager                       // 异步处理
│       └── security                      // 权限控制
│       └── web                           // 前端控制
